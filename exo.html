Configuration

Pour faire fonctionner notre application web sur votre navigateur, modifiez le fichier web.xml du dossier WEB-INF/, comme ci-dessous

<context-param>
<param-name>db.driver</param-name>
<param-value>org.postgresql.Driver</param-value>
</context-param>

<context-param>
<param-name>db.url</param-name>
<param-value>jdbc:postgresql://psqlserv/</param-value>
</context-param>

<context-param>
<param-name>db.base</param-name>
<param-value>da2i</param-value>
</context-param>

<context-param>
<param-name>db.username</param-name>
<param-value>leleuj</param-value>
</context-param>

<context-param>
<param-name>db.password</param-name>
<param-value>moi</param-value>
</context-param>
